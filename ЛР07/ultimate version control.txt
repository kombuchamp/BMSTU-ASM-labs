	;// и без лидирующих нулей (использовать команды сдвига).
	proc2 PROC
		mov bx, x ;//  число
		mov cx, 16 ;// –азмер массива 
		mov si,0 ;// »ндекс в массиве вывода
		
		jmp output
		counter_decrement:
			dec cx
		output:
			shl bx, 1
			setc al ;// al = CF ? 1 : 0
			; ѕровер€ем, если в переносе 0
			cmp al, 0
			jz check_if_preceding_zero
			jmp actual_output
			; ѕровер€ем, если это лидирующий ноль
			check_if_preceding_zero:
				cmp si, 0
				jz counter_decrement
			actual_output:
				mov arr[si],al
				inc si
		loop output



		ret
	proc2 ENDP
